create or replace FUNCTION FNBONUS(EMP IN NUMBER)
RETURN NUMBER IS 
  SAL EMPLOYEES.SALARY%TYPE;
  COM EMPLOYEES.COMMISSION_PCT%TYPE;
  BONUS NUMBER;
BEGIN 
   SELECT SALARY,COMMISSION_PCT INTO SAL,COM FROM EMPLOYEES WHERE EMPLOYEE_ID= EMP;
   
   
   IF COM IS NULL THEN
    
    BONUS := SAL*0.10 ;
    
    ELSE 
    
    BONUS := SAL * COM ;
    
    END IF;
    
   RETURN BONUS;
